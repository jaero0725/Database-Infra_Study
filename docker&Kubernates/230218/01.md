        # docker_server for docker
        config.vm.define "docker_ubuntu" do |cfg|
                cfg.vm.box = "generic/ubuntu2204"
                cfg.vm.provider "virtualbox" do |vb|
                        vb.name = "docker"
                        vb.cpus = 2
                        vb.memory = 2048
                        vb.gui = false
                end
                cfg.vm.host_name = "ubuntu.example.com"

        end
